<PROG> -> PROGRAM id <DEC> BEGIN <key_words> END.
<DEC> -> VAR <idlist> : type
<idlist> -> id <A>
<A> -> , <idlist> | @
<key_words> -> <key_word> <B>
<B> -> ; <C> | @
<C> -> <key_words> | @
<key_word> -> <IF> | <ASSIGNMENT> | <WHILE>
<IF> -> IF ( <condition> ) BEGIN <key_words> END
<condition> -> <Expr> more_less_operation <Expr>
<id_or_num> -> num | id
<ASSIGNMENT> -> id = <Expr>
<WHILE> -> WHILE ( <condition> ) BEGIN <key_words> END
<Expr> -> <E>
<E> -> <T> <G>
<G> -> + <T> <G> | @
<T> -> <F> <U>
<U> -> * <F> <U> | <F> | @
<F> -> ( <E> ) | - <F> | id | num
